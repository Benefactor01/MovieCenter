39          Parent root = loader.load();
40          guiStage.setScene(new Scene(root));
41  
42          //Az új ablakot (register.fxml) mozgathatóvá teszi
43  
44          guiStage.getScene().setOnMousePressed(new EventHandler<MouseEvent>() {
45              @Override
46              public void handle(MouseEvent event) {
47                  xOffset = event.getSceneX();
48                  yOffset = event.getSceneY();
49              }
50          });
51          guiStage.getScene().setOnMouseDragged(new EventHandler<MouseEvent>() {
52              @Override
53              public void handle(MouseEvent event) {
54                  guiStage.setX(event.getScreenX() - xOffset);
55                  guiStage.setY(event.getScreenY() - yOffset);
56              }
57          });
58      }

66      @FXML
67      private void signin_SendSignIn() {
68  
69          if(usernamesignin.getLength() > 15){
70              labelsignin.setTextFill(Color.DARKRED);
71              labelsignin.setStyle("-fx-font-weight: bold; -fx-font-size: 24;");
72              labelsignin.setText("A felhasználónév túl hosszú!");
73          }
74          if(passwordsignin.getLength() > 20){
75              labelsignin.setTextFill(Color.DARKRED);
76              labelsignin.setStyle("-fx-font-weight: bold; -fx-font-size: 24;");
77              labelsignin.setText("A jelszó túl hosszú!");
78          }
79  
80          if(passwordsignin.getText().trim().isEmpty()){
81              labelsignin.setTextFill(Color.DARKRED);
82              labelsignin.setStyle("-fx-font-weight: bold; -fx-font-size: 24;");
83              labelsignin.setText("A jelszó nem maradhat üres!");
84          }
85          if(usernamesignin.getText().trim().isEmpty()){
86              labelsignin.setTextFill(Color.DARKRED);
87              labelsignin.setStyle("-fx-font-weight: bold; -fx-font-size: 24;");
88              labelsignin.setText("Üres felhasználónév!");
89          }
90          if(!usernamesignin.getText().trim().isEmpty() && !passwordsignin.getText().trim().isEmpty() && usernamesignin.getLength() <= 15 && passwordsignin.getLength() <= 20){
91              if(UserController.login(usernamesignin.getText(), passwordsignin.getText()) == -2){
92                  labelsignin.setTextFill(Color.DARKRED);
93                  labelsignin.setStyle("-fx-font-weight: bold; -fx-font-size: 24;");
94                  labelsignin.setText("Nincs ilyen felhasználó!");
95              }
96              if(UserController.login(usernamesignin.getText(), passwordsignin.getText()) == -1){
97                  labelsignin.setTextFill(Color.DARKRED);
98                  labelsignin.setStyle("-fx-font-weight: bold; -fx-font-size: 24;");
99                  labelsignin.setText("Hibás jelszó!");
100             }
101             if(UserController.login(usernamesignin.getText(), passwordsignin.getText()) == 1){
102                 labelsignin.setTextFill(Color.LIMEGREEN);
103                 labelsignin.setStyle("-fx-font-weight: bold; -fx-font-size: 24;");
104                 labelsignin.setText("Sikeres bejelentkezés!");
105 
106                 //Időzítő 2.5 másodpercig hogy látható legyen a bejelentkezés sikeres felirat!
107 
108                 Timeline timeline = new Timeline(new KeyFrame(
109                         Duration.millis(2500),
110                         ae -> {
111                             FXMLLoader belep = new FXMLLoader(getClass().getClassLoader().getResource("view/Belepve.fxml"));
112                             Parent root = null;
113                             try {
114                                 root = belep.load();
115                             } catch (IOException e) {
116                                 e.printStackTrace();
117                             }
118 
119                             //Belépett felhasználó továbbküldése ControllerOfInside-nek
120                             ControllerOfInside transfer1 = belep.getController();
121                             transfer1.transferMessage(usernamesignin.getText());
122 
123                             guiStage.close();
124                             guiStage.setScene(new Scene(root, 800, 600));
125                             guiStage.show();
126 
127 
128 
129                             guiStage.getScene().setOnMousePressed(new EventHandler<MouseEvent>() {
130                                 @Override
131                                 public void handle(MouseEvent event) {
132                                     xOffset = event.getSceneX();
133                                     yOffset = event.getSceneY();
134                                 }
135                             });
136                             guiStage.getScene().setOnMouseDragged(new EventHandler<MouseEvent>() {
137                                 @Override
138                                 public void handle(MouseEvent event) {
139                                     guiStage.setX(event.getScreenX() - xOffset);
140                                     guiStage.setY(event.getScreenY() - yOffset);
141                                 }
142                             });
143                             // Main.getStage().setMaximized(true);
144                         }));
145                 timeline.play();
146             }
147         }
148     }
